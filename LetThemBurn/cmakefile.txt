CC	= g++

NAME	= bin

RM	= rm -f

CFLAGS	+= -W -Wall -Wextra -Wfatal-errors

SRC	=	main.cpp \
		Peon.cpp \
		Sorcerer.cpp \
		Victim.cpp


OBJS	= $(SRC:.cpp=.o) #

all: $(NAME)

$(NAME): $(OBJS)
	 $(CC) $(OBJS) $(CFLAGS) -o $(NAME)

clean:
	$(RM) $(OBJS)

fclean: clean
	$(RM) $(NAME)

re: fclean all

.PHONY:re clean fclean all
